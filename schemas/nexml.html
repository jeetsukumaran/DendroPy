Tree.get¶

(method reference)

    path="path/to/file",
    schema="nexml",
    label=None,
    taxon_namespace=None,
    collection_offset=None,
    tree_offset=None,
    default_namespace="http://www.nexml.org/2009",
    case_sensitive_taxon_labels=False,
    ignore_unrecognized_keyword_arguments=False,
    )
